country_event = {
	# Event to leave a faction because of a revolution
	id = r56i.event.10
	title = r56i.event.10.title
	desc = r56i.event.10.description
	picture = GFX_report_event_spain_civil_war_volunteers_01
	
	is_triggered_only = yes
	
	option = {
		name = r56i.event.10.a
		ai_chance = {
			factor = 1
			modifier = {
				#value security of being in a faction
				OR = {
					any_country = {
						has_government = ROOT
						is_faction_leader = yes
						is_in_faction_with = ROOT					
					}
					has_government = neutrality
					AND = {
						original_tag = NOR
						has_government = democratic
						any_country = {
							has_government = neutrality
							is_faction_leader = yes
							is_in_faction_with = NOR					
						}						
					}
				}
				factor = 0
			}			
		}
		hidden_effect = {
			news_event = { id = r56i.event.11 }
		}
		every_country = {
			limit ={
				has_war_with = ROOT
				OR = {
					AND = {
						is_in_faction = no
						has_government = ROOT
					}						
					AND = {
						is_faction_leader = yes
						has_government = ROOT
					}
					AND = {
						is_in_faction = yes
						any_country = {
							is_faction_leader = yes
							is_in_faction_with = PREV
							has_government = ROOT
						}
					}
				}
			}
			white_peace = ROOT	
		}
		every_country = {
			limit = {
				is_in_faction_with = ROOT
			}
			add_opinion_modifier = {
				modifier = faction_traitor
				target = ROOT
				days = 730
			}
			add_opinion_modifier = {
				modifier = faction_traitor_trade
				target = ROOT
				days = 730
			}
		}	
		leave_faction = yes
	}

	option = {
		name = r56i.event.10.b
		ai_chance = {
			factor = 0.2
		}		
	}
}
